function calculateROI() {
    const devicePrice = parseInt(document.getElementById('devicePrice').value);
    const rentalPrice = parseInt(document.getElementById('rentalPrice').value);
    const rentalDaysMonth = parseInt(document.getElementById('rentalDaysMonth').value);
    const maintenanceCost = parseInt(document.getElementById('maintenanceCost').value);
    
    const monthlyIncome = rentalPrice * rentalDaysMonth;
    const monthlyProfit = monthlyIncome - maintenanceCost;
    const paybackMonths = Math.ceil(devicePrice / monthlyProfit);
    const yearlyProfit = monthlyProfit * 12;
    const roi = ((yearlyProfit / devicePrice) * 100).toFixed(1);
    
    let html = '<h4>Анализ окупаемости:</h4>';

            if (saved) {

                document.querySelectorAll('.checklist input[type="checkbox"]').forEach((checkbox, index) => {

                });
            }

        // Calculator Functions
        function calculateDMXAddressing() {
            const devices = parseInt(document.getElementById('dmxDevices').value);
            const channels = parseInt(document.getElementById('dmxChannels').value);
            const start = parseInt(document.getElementById('dmxStart').value);
            
            let html = '<h4>Результаты адресации:</h4>';
            html += '<table><tr><th>Прибор</th><th>Старт</th><th>Конец</th><th>Universe</th></tr>';
            
            let currentAddress = start;
            
            for (let i = 1; i <= devices; i++) {
                const endAddress = currentAddress + channels - 1;
                const universe = Math.ceil(currentAddress / 512);
                const universeEnd = Math.ceil(endAddress / 512);
                
                html += `<tr>
                    <td>Прибор ${i}</td>
                    <td>${currentAddress}</td>
                    <td>${endAddress}</td>
                    <td>Universe ${universe}${universe !== universeEnd ? '-' + universeEnd : ''}</td>
                </tr>`;
                
                if (universeEnd > universe) {
                    html += '<tr><td colspan="4" style="color: var(--danger-border);">⚠️ Прибор пересекает границу Universe!</td></tr>';
                }
                
                currentAddress = endAddress + 1;
            }
            
            html += '</table>';
            html += `<p><strong>Всего используется каналов:</strong> ${devices * channels}</p>`;
            html += `<p><strong>Требуется Universes:</strong> ${Math.ceil((start + devices * channels - 1) / 512)}</p>`;
            
            document.getElementById('dmxAddressingResult').innerHTML = html;
            
            // Update DMX grid visualization
            updateDMXGrid(start, devices, channels);
        }

        function updateDMXGrid(start, devices, channels) {
            const grid = document.getElementById('dmxUniverseGrid');
            if (!grid) return;
            
            grid.innerHTML = '<h4>Визуализация DMX Universe (первые 128 каналов):</h4>';
            
            const usedChannels = new Set();
            let currentAddress = start;
            
            for (let i = 0; i < devices; i++) {
                for (let j = 0; j < channels; j++) {
                    if (currentAddress <= 512) {
                        usedChannels.add(currentAddress);
                        currentAddress++;
                    }
                }
            }
            
            for (let i = 1; i <= 128; i++) {
                const channel = document.createElement('div');
                channel.className = 'dmx-channel';
                if (usedChannels.has(i)) {
                    channel.classList.add('used');
                }
                channel.textContent = i;
                channel.onclick = function() {
                    this.classList.toggle('selected');
                };
                grid.appendChild(channel);
            }
        }

        function calculatePowerLoad() {
            const devices = parseInt(document.getElementById('calcDevices').value);
            const watts = parseInt(document.getElementById('calcWatts').value);
            const safety = parseFloat(document.getElementById('calcSafety').value);
            
            const totalWatts = devices * watts;
            const withSafety = totalWatts * safety;
            const current230 = withSafety / 230 / 0.9; // cos φ = 0.9
            const current400 = withSafety / (400 * 1.73 * 0.9);
            
            let html = '<h4>Результаты расчета:</h4>';
            html += `<p><strong>Суммарная мощность:</strong> ${(totalWatts/1000).toFixed(2)} кВт</p>`;
            html += `<p><strong>С учетом запаса:</strong> ${(withSafety/1000).toFixed(2)} кВт</p>`;
            html += '<h5>Однофазная сеть 230В:</h5>';
            html += `<p>Ток: ${current230.toFixed(1)} А</p>`;
            html += `<p>Рекомендуемый автомат: ${Math.ceil(current230/10)*10} А</p>`;
            html += '<h5>Трехфазная сеть 400В:</h5>';
            html += `<p>Ток: ${current400.toFixed(1)} А</p>`;
            html += `<p>Рекомендуемый разъем: CEE ${current400 < 16 ? '16A' : current400 < 32 ? '32A' : current400 < 63 ? '63A' : '125A'}</p>`;
            
            document.getElementById('powerLoadResult').innerHTML = html;
        }

        function calculateRigging() {
            const weight = parseFloat(document.getElementById('loadWeight').value);
            const points = parseInt(document.getElementById('loadPoints').value);
            const angle = parseFloat(document.getElementById('slingAngle').value);
            
            const angleRad = angle * Math.PI / 180;
            const loadPerPoint = weight / points;
            const slingLoad = loadPerPoint / Math.cos(angleRad / 2);
            const safetyFactor = 10;
            const requiredWLL = slingLoad * safetyFactor / 1000; // Convert to tons
            
            let html = '<h4>Результаты расчета:</h4>';
            html += `<p><strong>Нагрузка на точку:</strong> ${loadPerPoint.toFixed(1)} кг</p>`;
            html += `<p><strong>Нагрузка на стропу:</strong> ${slingLoad.toFixed(1)} кг</p>`;
            html += `<p><strong>Коэффициент увеличения от угла:</strong> ${(slingLoad/loadPerPoint).toFixed(2)}</p>`;
            html += `<p><strong>Требуемая WLL стропы (1:10):</strong> ${requiredWLL.toFixed(2)} т</p>`;
            
            if (angle > 120) {
                html += '<div class="alert alert-danger">⚠️ Угол слишком большой! Не превышайте 120°</div>';
            }
            
            document.getElementById('riggingResult').innerHTML = html;
        }

        function calculateViewingDistance() {
            const pitch = parseFloat(document.getElementById('pixelPitch').value);
            
            const minDistance = pitch * 1;
            const optDistance = pitch * 3;

            html += `<li>Различимость пикселей теряется после ${maxDistance.toFixed(1)}м</li>`;

        }

        function checkQuizAnswer(element, isCorrect) {
            const options = element.parentElement.querySelectorAll('.quiz-option');
            
            // Disable all options
            options.forEach(opt => {
                opt.style.pointerEvents = 'none';
            });
            
            if (isCorrect) {
                element.classList.add('correct');
                quizScore++;
            } else {
                element.classList.add('incorrect');
                // Show correct answer
                options.forEach(opt => {
                    if ((opt.textContent === '512' && element.parentElement.parentElement.querySelector('h4').textContent.includes('DMX')) ||
                        (opt.textContent === '1:10' && element.parentElement.parentElement.querySelector('h4').textContent.includes('риггинге')) ||
                        (opt.textContent === '44 Hz' && element.parentElement.parentElement.querySelector('h4').textContent.includes('частота'))) {
                        opt.classList.add('correct');
                    }
                });
            }
            
            quizAnswered++;
            
            // Show results if all questions answered
            if (quizAnswered >= 3) {
                const resultDiv = document.getElementById('quizResult');
                const scoreText = document.getElementById('quizScore');
                scoreText.textContent = `Вы ответили правильно на ${quizScore} из 3 вопросов (${Math.round(quizScore/3*100)}%)`;
                resultDiv.classList.add('show');
            }
        }
